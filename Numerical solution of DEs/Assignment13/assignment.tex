\problem{1}{Prove 4.2.16}
\proof{Proof}{
The truncation error at point $(t_n, x_i)$ is
\[
\begin{aligned}
R_i^n &= \frac{u(t_{n+1}, x_i)-u(t_{n-1}, x_i)}{2\tau} - a\Delta_h u(t_n, x_i) - f(t_n, x_i) \\
&= R_t + R_x,
\end{aligned}
\]
where
\[
\begin{aligned}
R_t &= \frac{u(t_{n+1}, x_i)-u(t_{n-1}, x_i)}{2\tau} - \frac{\partial u}{\partial t}(t_n, x_i) \\
&= \frac{\tau^2}{6}\frac{\partial^3u}{\partial t^3}(t_n, x_i)+O(\tau^4), \\
R_x &= -a\Delta_h u_i^n + a\frac{\partial^2u}{\partial x^2}(t_n, x_i) \\
&= -\frac{ah^2}{12}\frac{\partial^4u}{\partial x^4}(t_n, x_i)+O(h^4)
\end{aligned}
\]
Adding the two terms we can get (4.2.16).
}

\problem{2}{Problem 1, Page 195}
\solution{The schemes}{
The iteration of $\theta$ scheme is
\[
(\mx{I} - a\tau\theta\Delta_h)\mx{u}^{n+1} = (\mx{I} + a\tau(1-\theta)\Delta_h)\mx{u}^{n} + \tau(\theta \mx{f}^{n+1}+\mx{f}^n),
\]
where $\mx{u}$ and $\mx{f}$ are vectors of grid points.

The exact solution is
\[
u(t, x) = \sin\pi xe^{-\pi^2t}.
\]
}
